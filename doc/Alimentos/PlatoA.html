def huella_nut
  cal = calories
  gas = plato_ambiente

  if cal < 670
    cal = 1
  elsif cal < 830
    cal = 2
  else
    cal = 3
  end

  if gas < 800
    gas = 1
  elsif gas < 1200
    gas = 2
  else
    gas = 3
  end

  (cal + gas) / 2.round(2)
end

def plato_ambiente
  aux = []
  for i in 0..@grams.length() - 1
    aux[i] = @grams[i] / 100
  end
  x = 0
  for i in 0..@receip.tam - 1
    x += @receip.lista[i].value.gei * aux[i]
  end
  x.round(2)
end
