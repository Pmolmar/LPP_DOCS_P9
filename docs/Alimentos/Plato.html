def calories
  aux = []
  for i in 0..@grams.length() - 1
    aux[i] = @grams[i] / 100
  end

  cal = 0
  for i in 0..@grams.length() - 1
    cal += @receip.lista[i].value.energi * aux[i]
  end

  (cal).round(2)
end

def carPercent
  aux = 0
  for i in 0..@receip.tam - 1
    aux += @receip.lista[i].value.car
  end

  x = []
  for i in 0..@receip.tam - 1
    x[i] = (@receip.lista[i].value.car / aux).round(2)
  end
  x
end
