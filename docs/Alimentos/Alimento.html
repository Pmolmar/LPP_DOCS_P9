def impactMen(x)
  [((energi / 3000) * x / 100).round(2), ((@pro / 54) * x / 100).round(2)]
end

def impactWom(x)
  [((energi / 2300) * x / 100).round(2), ((@pro / 41) * x / 100).round(2)]
end
